OUTPUT_FORMAT("elf32-i386")
ENTRY(start)
entry = 0x00100000;
SECTIONS {
	.text entry : AT (entry) {
		code = .;
		*(.text)
		. = ALIGN(4096);
	}
	textEnd = .;

	.data : AT(entry + (data - code)) {
		data = .;
		*(.data)
		*(.rodata)
		. = ALIGN(4096);
	}
	dataEnd = .;

	.bss : AT(entry + (bss - code)){
		bss = .;
		*(.common)
		*(.bss)
		. = ALIGN(4096);
	}
	. = ALIGN(4096);
	bssEnd = .;
	end = .;
}
INPUT(kernel.o startup.o console.o multiboot.o gdt.o idt.o isr.o irq.o x86.o timer.o mem.o lib/libc.a)
OUTPUT(kernel.bin)
